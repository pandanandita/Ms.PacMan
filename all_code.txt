//Nandita Panda//Feb 21, 2026//Project 3import java.lang.Math;import java.util.ArrayList;import java.util.Iterator;public class Model {    private int scrollY;    //scrolling logic    private ArrayList<Tile> tiles;      //array to hold tiles    msPacman mspacman;    public Model() {        this.scrollY = 0;        this.tiles = new ArrayList<>();        this.mspacman = new msPacman(200, 300);        try {            this.load("map.json");        } catch (Exception e) {            System.out.println("No map found");        }    }    public void update() {        mspacman.update();        Iterator<Tile> it = tiles.iterator();        while(it.hasNext()) {            Tile t = it.next();            if(detectCollision(mspacman, t)) {                mspacman.getOutOfTile(t);            }        }    }    private boolean detectCollision(msPacman p, Tile t) {        if (p.getRight() < t.getLeft()) { return false; }        if (p.getLeft() > t.getRight()) { return false; }        if (p.getBottom() < t.getTop()) { return false; }        if (p.getTop() > t.getBottom()) { return false; }        return true;    }    public void clear() {        tiles.clear();    }    //creates new tile object and adds to list    public void addTile(int x, int y) {        Tile tile = new Tile(x, y);        tiles.add(tile);    }    // check for exisiting tiles    public boolean checkTile(int x, int y) {        for (int i = 0; i < 4; i++) {                for (int j = 0; j < 3; j++) {                    images[i][j] = View.loadImage("images/mspacman" + count + ".png");                    count++;                }            }        }    }    public void update() {        if (this.x < -50) {            this.x = 810;        } else if (this.x > 810) {            this.x = -50;        }        if (this.y < 0) {            this.y = 0;        }        if (this.y > 510) {            this.y = 510;        }    }    public void setPreviousPosition() {        this.px = x;        this.py = y;    }    public void scrollLeft() {        direction = 0;        x -= speed;        nextFrame();    }    public void scrollUp() {        direction = 1;        y -= speed;        nextFrame();    }    public void scrollRight() {        direction = 2;        x += speed;        nextFrame();    }    public void scrollDown() {        direction = 3;        y += speed;        nextFrame();    }    private void nextFrame() {        currentFrame++;        if (currentFrame >= 3) {            currentFrame = 0;        }    }    public void draw(Graphics g, int scrollPos) {        g.drawImage(images[direction][currentFrame], x, y - scrollPos, w, h, null);    }    public int getY() { return y; }    public int getPreviousLeft() { return px; }    public int getPreviousRight() { return px + w; }    public int getPreviousTop() { return py; }    public int getPreviousBottom() { return py + h; }    public int getLeft() { return x; }    public int getRight() { return x + w; }    public int getTop() { return y; }    public int getBottom() { return y + h; }    @Override    public String toString() {        return "MsPacman (x,y) = (" + x + ", " + y + "), w = " + w + ", h = " + h;    }    public void getOutOfTile(Tile t) {        //colliding from left        if (getRight() >= t.getX() && getPreviousRight() <= t.getX()) {            x = t.getX() - w - 1;        }        //colliding from right        else if (getLeft() <= (t.getX() + t.getW()) && getPreviousLeft() >= (t.getX() + t.getW())) {            x = t.getX() + t.getW() + 1;        }        //colliding from top        if (getBottom() >= t.getY() && getPreviousBottom() <= t.getY()) {            y = t.getY() - h - 1;        }        //colliding from bottom        else if (getTop() <= (t.getY() + t.getH()) && getPreviousTop() >= (t.getY() + t.getH())) {            y = t.getY() + t.getH() + 1;        }    }}

//Nandita Panda//Feb 21, 2026//Project 3import java.awt.Graphics;import java.awt.image.BufferedImage;public class Tile {    private int x, y, w, h;    public static BufferedImage image = null;    public  Tile(int x, int y) {        this.x = x;        this.y = y;        this.w = 40;        this.h = 40;        load_image();    }    // rebuilds tile object from json object    // used to load map from a file    //converts to json    public Tile(Json ob) {        this.x = (int)ob.getLong("x");        this.y = (int)ob.getLong("y");        this.w = (int)ob.getLong("w");        this.h = (int)ob.getLong("h");        load_image();    }    // converts current tile object into json    // saves tile's data to file    //converted from json    public Json marshal() {        Json ob = Json.newObject();        ob.add("x", this.x);        ob.add("y", this.y);        ob.add("w", this.w);        ob.add("h", this.h);        return ob;    }    private void load_image() {        if (image == null) {            image = View.loadImage("images/tile1.jpg");        }    }    public void draw(Graphics g, int scrollPos) {        g.drawImage(image, x, y - scrollPos, w, h, null);    }    //returns true if exists at that location    public boolean isAt(int x, int y) {        return (this.x == x && this.y == y);    }    @Override    public String toString() {            return "Tile (x,y) = (" + x + ", " + y + "), w = " + w + ", h = " + h;    }    public int getLeft() { return x; }    public int getRight() { return x + w; }    public int getTop() { return y; }    public int getBottom() { return y + h; }    public int getX() {        return x;    }    public int getY() {        return y;    }    public int getW() {        return w;    }    public int getH() {        return h;    }    public void setX(int x) {        this.x = x;    }    public void setY(int y) {        this.y = y;    }    public void setW(int w) {        this.w = w;    }    public void setH(int h) {        this.h = h;    }}

//Nandita Panda//Feb 21, 2026//Project 3import javax.swing.JPanel;import java.awt.Graphics;import java.awt.image.BufferedImage;import javax.imageio.ImageIO;import java.io.IOException;import java.io.File;import javax.swing.JButton;import java.awt.Color;public class View extends JPanel {    private Model model;    private Controller controller;    private int scrollPos;    public View(Controller c, Model m) {        this.model = m;        this.controller = c;        c.setView(this);    }    public static BufferedImage loadImage(String filename) {        try {            return ImageIO.read(new File(filename));        } catch (Exception e) {            e.printStackTrace(System.err);            System.exit(1);            return null;        }    }    //override paint method    public void paintComponent(Graphics g) {        //view needs to know how far world has scrolled            g.setColor(new Color(254, 189, 149));            g.fillRect(0, 0, this.getWidth(), this.getHeight());        this.scrollPos = model.getScrollY();        for(int i = 0; i < model.getTiles().size(); i++) {            model.getTiles().get(i).draw(g, scrollPos);        }        model.mspacman.draw(g, scrollPos);        //subtracting creates scrolling effect of camera moves down, objects drawn higher        //tile.getY() - scrollPos creates scroll effect. as scrollY increases, tiles move upward (visually)        //if editmode on, green box, if remove, red        if (controller.isEditMode()) {            if (controller.isAddMapItem()) {                g.setColor(Color.GREEN);            }            else {                g.setColor(Color.RED);            }            g.fillRect(0,0,100,100);            g.drawImage(Tile.image, 25, 25, 50, 50, null);        }    }}//scrollPos is part of world's state so in model (stores tile and scroll amount)//view needs it so it can shift tiles visually
